109 public void Parse(string[] args)

111 IOption? TryGetOption(string name) => options.Find(o => o.MatchingOptionName(name));

117 int GetOptionParamCount(IOption option, string[] args, int argIdx)
118 {
119 int result = 0;
120 while (result < option.ParameterAccept.MaxParamAmount)
121 {
122 int idx = argIdx + result + 1;
123 if (idx == args.Length || args[idx] == Delimiter || TryGetOption(args[idx]) != null)
124 {
125 if (result < option.ParameterAccept.MinParamAmount) throw new ParseException("Too few parameters passed to option");
126 else break;
127 }
128 result++;
129 }
130 return result;
131 }

136 int ParseOptions(string[] args)
137 {
138 bool IsPlainArg(string arg) => !arg.StartsWith("-");
139
140 int argIdx = 0;
141 while (argIdx < args.Length)
142 {
143 var currentArg = args[argIdx];
144
145 if (IsPlainArg(currentArg)) break;
146 if (currentArg == Delimiter)
147 {
148 argIdx++;
149 break;
150 }
151
152 var option = TryGetOption(currentArg);
153 if (option == null) throw new ParseException("Unrecognized option");
154
155 var paramCount = GetOptionParamCount(option, args, argIdx);
156 option.CallParse(args[(argIdx + 1)..(argIdx + paramCount + 1)]);
157 argIdx += paramCount + 1;
158 }
159
160 return argIdx;
161 }

163 void CheckPlainParamCounts(IArgument[] orderedArguments, int argsLength)
164 {
165 int min = orderedArguments
166 .Select(a => a.ParameterAccept.MinParamAmount)
167 .Aggregate(0, (acc, val) => acc + val);
168
169 int max = orderedArguments
170 .Select(a => a.ParameterAccept.MaxParamAmount)
171 .Aggregate(0, (acc, val) => acc + val);
172
173 if (argsLength < min || argsLength > max) throw new ParseException("arguments are incompatible");
174 }

176 int GetPlainParamCount(int plainArgIdx, int argsCount)
177 {
178 var OrderedArguments = GetArgumentOrder();
179 var currentArg = OrderedArguments[plainArgIdx];
180
181 if (!currentArg.ParameterAccept.IsVariadic)
182 return currentArg.ParameterAccept.MinParamAmount;
183
184
185 var argCountRequiredByFollowingArgs = OrderedArguments[(plainArgIdx + 1)..]
186 .Select(a => a.ParameterAccept.MinParamAmount)
187 .Aggregate(0, (acc, val) => acc + val);
188
189 return argsCount - argCountRequiredByFollowingArgs;
190 }

192 void ParsePlainArgs(string[] args, int argIdx, IArgument[] orderedArguments)
193 {
194 for (var plainArgIdx = 0; plainArgIdx < orderedArguments.Length; plainArgIdx++)
195 {
196 var remainingArgumentsCount = args.Length - argIdx;
197 var currentArg = orderedArguments[plainArgIdx];
198 var valCount = GetPlainParamCount(plainArgIdx, remainingArgumentsCount);
199
200 string[] vals = args[argIdx..(argIdx + valCount)];
201
202 currentArg.CallParse(vals);
203 argIdx += valCount;
204 }
205 }
